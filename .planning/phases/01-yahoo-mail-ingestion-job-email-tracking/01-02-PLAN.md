    **normalizeSubject(subject: string): string**
    Strip leading Re:/Fwd:/fw: prefixes (case-insensitive, handle chained Re: Re:), then trim and lowercase.
    ```typescript
    export function normalizeSubject(subject: string): string {
      return subject.replace(/^(re|fwd|fw)\s*:\s*/gi, '').trim().toLowerCase();
    }
    ```
    Note: This regex only strips ONE prefix. For chained "Re: Re: X" the test expects both stripped. Use a loop or recursive replace:
    ```typescript
    export function normalizeSubject(subject: string): string {
      let s = subject;
      let prev: string;
      do {
        prev = s;
        s = s.replace(/^(re|fwd|fw)\s*:\s*/i, '').trim();
      } while (s !== prev);
      return s.toLowerCase();
    }
    ```
